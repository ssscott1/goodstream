// ═══ STATE ═══
let isLoggedIn = false;
let categories = ['Specifications', 'Data Sheets', 'Requirements'];
let files = [];
let activeFilter = 'All';

function saveState() {
  try {
    const meta = files.map(f => ({ id: f.id, name: f.name, category: f.category, size: f.size, date: f.date }));
    localStorage.setItem('gs_categories', JSON.stringify(categories));
    localStorage.setItem('gs_files_meta', JSON.stringify(meta));
    files.forEach(f => { if (f.dataUrl) localStorage.setItem('gs_file_' + f.id, f.dataUrl); });
  } catch (e) { console.warn('Storage limit reached'); }
}
function loadState() {
  try {
    const c = localStorage.getItem('gs_categories');
    const m = localStorage.getItem('gs_files_meta');
    if (c) categories = JSON.parse(c);
    if (m) { files = JSON.parse(m).map(f => ({ ...f, dataUrl: localStorage.getItem('gs_file_' + f.id) || null })); }
  } catch (e) {}
}
loadState();

// ═══ AUTH ═══
function openLoginModal() {
  const m = document.getElementById('loginModal');
  m.style.display = 'flex';
  requestAnimationFrame(() => m.classList.add('active'));
  document.getElementById('loginUser').focus();
}
function closeLoginModal() {
  const m = document.getElementById('loginModal');
  m.classList.remove('active');
  setTimeout(() => m.style.display = 'none', 300);
  document.getElementById('loginError').classList.remove('show');
  document.getElementById('loginUser').value = '';
  document.getElementById('loginPass').value = '';
}
function attemptLogin() {
  const u = document.getElementById('loginUser').value.trim();
  const p = document.getElementById('loginPass').value;
  if (u === 'admin' && p === 'Password1$') {
    isLoggedIn = true;
    closeLoginModal();
    document.getElementById('portalPublic').style.display = 'none';
    document.getElementById('adminDashboard').classList.add('active');
    renderAdminCategories(); renderFilesTable(); updateUploadCategorySelect();
    showToast('Logged in successfully', 'success');
  } else {
    document.getElementById('loginError').classList.add('show');
  }
}
function logout() {
  isLoggedIn = false;
  document.getElementById('adminDashboard').classList.remove('active');
  document.getElementById('portalPublic').style.display = 'block';
  renderPublicView();
  showToast('Signed out', 'success');
}

// ═══ CATEGORIES ═══
function showAddCategory() { document.getElementById('addCatBtn').style.display = 'none'; document.getElementById('addCatForm').classList.add('show'); document.getElementById('newCatInput').focus(); }
function hideAddCategory() { document.getElementById('addCatBtn').style.display = ''; document.getElementById('addCatForm').classList.remove('show'); document.getElementById('newCatInput').value = ''; }
function addCategory() {
  const name = document.getElementById('newCatInput').value.trim();
  if (!name) return;
  if (categories.includes(name)) { showToast('Category already exists', 'error'); return; }
  categories.push(name); hideAddCategory();
  renderAdminCategories(); updateUploadCategorySelect(); saveState();
  showToast('Category "' + name + '" added', 'success');
}
function removeCategory(cat) {
  if (!confirm('Delete category "' + cat + '"? Files will be moved to Uncategorised.')) return;
  files.forEach(f => { if (f.category === cat) f.category = 'Uncategorised'; });
  categories = categories.filter(c => c !== cat);
  if (activeFilter === cat) activeFilter = 'All';
  renderAdminCategories(); renderFilesTable(); updateUploadCategorySelect(); saveState();
  showToast('Category deleted', 'success');
}
function setFilter(cat) {
  activeFilter = cat;
  if (isLoggedIn) { renderAdminCategories(); renderFilesTable(); } else { renderPublicView(); }
}
function esc(s) { return s.replace(/'/g, "\\'"); }

function renderAdminCategories() {
  const bar = document.getElementById('adminCategoryBar');
  const counts = {}; files.forEach(f => counts[f.category] = (counts[f.category] || 0) + 1);
  let h = '<span class="category-tag ' + (activeFilter === 'All' ? 'active' : '') + '" onclick="setFilter(\'All\')">All <span class="cat-count">' + files.length + '</span></span>';
  categories.forEach(cat => {
    h += '<span class="category-tag ' + (activeFilter === cat ? 'active' : '') + '" onclick="setFilter(\'' + esc(cat) + '\')">' + cat + ' <span class="cat-count">' + (counts[cat] || 0) + '</span> <span class="cat-remove" onclick="event.stopPropagation();removeCategory(\'' + esc(cat) + '\')" title="Delete">✕</span></span>';
  });
  if (counts['Uncategorised']) h += '<span class="category-tag ' + (activeFilter === 'Uncategorised' ? 'active' : '') + '" onclick="setFilter(\'Uncategorised\')">Uncategorised <span class="cat-count">' + counts['Uncategorised'] + '</span></span>';
  bar.innerHTML = h;
}
function updateUploadCategorySelect() {
  const s = document.getElementById('uploadCategorySelect');
  s.innerHTML = '<option value="Uncategorised">Uncategorised</option>';
  categories.forEach(c => s.innerHTML += '<option value="' + c + '">' + c + '</option>');
}

// ═══ FILE UPLOAD ═══
const uploadZone = document.getElementById('uploadZone');
const fileInput = document.getElementById('fileInput');
uploadZone.addEventListener('dragover', e => { e.preventDefault(); uploadZone.classList.add('dragover'); });
uploadZone.addEventListener('dragleave', () => uploadZone.classList.remove('dragover'));
uploadZone.addEventListener('drop', e => { e.preventDefault(); uploadZone.classList.remove('dragover'); handleFiles(e.dataTransfer.files); });
fileInput.addEventListener('change', () => { handleFiles(fileInput.files); fileInput.value = ''; });

function handleFiles(list) {
  const cat = document.getElementById('uploadCategorySelect').value;
  Array.from(list).forEach(file => {
    if (file.type !== 'application/pdf') { showToast(file.name + ' is not a PDF', 'error'); return; }
    if (file.size > 10485760) { showToast(file.name + ' exceeds 10MB', 'error'); return; }
    const reader = new FileReader();
    reader.onload = e => {
      files.push({ id: Date.now().toString(36) + Math.random().toString(36).substr(2, 5), name: file.name, category: cat, size: formatSize(file.size), date: new Date().toLocaleDateString('en-AU', { day: '2-digit', month: 'short', year: 'numeric' }), dataUrl: e.target.result });
      renderAdminCategories(); renderFilesTable(); saveState();
      showToast(file.name + ' uploaded', 'success');
    };
    reader.readAsDataURL(file);
  });
}
function formatSize(b) { return b < 1024 ? b + ' B' : b < 1048576 ? (b / 1024).toFixed(1) + ' KB' : (b / 1048576).toFixed(1) + ' MB'; }

function deleteFile(id) {
  if (!confirm('Delete this file?')) return;
  files = files.filter(f => f.id !== id);
  try { localStorage.removeItem('gs_file_' + id); } catch(e) {}
  renderAdminCategories(); renderFilesTable(); saveState();
  showToast('File deleted', 'success');
}
function downloadFile(id) {
  const f = files.find(x => x.id === id);
  if (!f || !f.dataUrl) { showToast('File data not available', 'error'); return; }
  const a = document.createElement('a'); a.href = f.dataUrl; a.download = f.name; a.click();
}

function renderFilesTable() {
  const tbody = document.getElementById('filesTableBody');
  const list = activeFilter === 'All' ? files : files.filter(f => f.category === activeFilter);
  if (!list.length) { tbody.innerHTML = '<tr><td colspan="5"><div class="empty-state"><div class="empty-state-icon">📂</div><p>No files' + (activeFilter !== 'All' ? ' in this category' : ' uploaded yet') + '.</p></div></td></tr>'; return; }
  tbody.innerHTML = list.map(f => '<tr><td><div class="file-name-cell"><div class="file-pdf-icon">PDF</div><span>' + f.name + '</span></div></td><td><span class="file-cat-badge">' + f.category + '</span></td><td>' + f.size + '</td><td>' + f.date + '</td><td style="text-align:right;"><button class="file-action-btn" onclick="downloadFile(\'' + f.id + '\')">⬇ Download</button><button class="file-action-btn delete" onclick="deleteFile(\'' + f.id + '\')">🗑 Delete</button></td></tr>').join('');
}

// ═══ PUBLIC VIEW ═══
function renderPublicView() {
  const bar = document.getElementById('publicCategoryBar');
  const grid = document.getElementById('publicFilesGrid');
  const cats = [...new Set(files.map(f => f.category))];
  if (!files.length) { bar.innerHTML = ''; } else {
    let h = '<span class="category-tag ' + (activeFilter === 'All' ? 'active' : '') + '" onclick="setFilter(\'All\')">All <span class="cat-count">' + files.length + '</span></span>';
    cats.forEach(c => { const n = files.filter(f => f.category === c).length; h += '<span class="category-tag ' + (activeFilter === c ? 'active' : '') + '" onclick="setFilter(\'' + esc(c) + '\')">' + c + ' <span class="cat-count">' + n + '</span></span>'; });
    bar.innerHTML = h;
  }
  const list = activeFilter === 'All' ? files : files.filter(f => f.category === activeFilter);
  if (!list.length) { grid.innerHTML = '<div class="empty-state" style="grid-column:1/-1;"><div class="empty-state-icon">📄</div><p>No documents ' + (activeFilter !== 'All' ? 'in this category' : 'uploaded yet') + '.</p></div>'; return; }
  grid.innerHTML = list.map(f => '<div class="public-file-card"><div class="public-file-pdf">PDF</div><div class="public-file-info"><div class="public-file-name">' + f.name + '</div><div class="public-file-meta">' + f.category + ' · ' + f.size + ' · ' + f.date + '</div></div><button class="public-download-btn" onclick="downloadFile(\'' + f.id + '\')" title="Download">⬇</button></div>').join('');
}

}
